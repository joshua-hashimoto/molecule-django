## docker-compose.yml

```yaml
version: "3"

services:
    web:
        build: .
        command: gunicorn core.wsgi -b 0.0.0.0:8000 # gunicorn対応
        env_file: .env
        volumes:
            - .:/app
        ports:
            - 8000:8000
        depends_on:
            - db
    db:
        image: postgres:11
        volumes:
            - postgres_data:/var/lib/postgresql/data/
        environment:
            - "POSTGRES_HOST_AUTH_METHOD=trust"

volumes:
    postgres_data:
```
